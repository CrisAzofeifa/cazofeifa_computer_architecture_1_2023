96template<typename FirstIface, typename... RestIfaces>
97class Connect : public ConnectBase {
98 public:
99 typedef std::tuple<typename FirstIface::ToC,
100 typename RestIfaces::ToC...> ifaces_type;
101 Connect() = default;
102 explicit Connect(const ConnectConfig &config) : config_(config) {}
103
104 // ConnectBase
105 bool set(const ConfObjectRef &o) override {
106 if (obj_ == o) {
107 return true;
108 }
109
110 if (!o) {
111 ifaces_ = ifaces_type();
112 obj_ = o;
113 return true;
114 }
115
116 try {
117 set_ifaces<FirstIface, RestIfaces...>(o);
118 } catch (const std::exception &e) {
119 SIM_LOG_INFO(1, o.object(), 0, "%s", e.what());
120 ifaces_ = ifaces_type();
121 return false;
122 }
123
124 obj_ = o;
125 return true;
126 }
127
135 template<typename T>
136 typename std::enable_if<sizeof...(RestIfaces) != 0,
137 const typename T::ToC>::type
138 iface() const {
139 return std::get<typename T::ToC>(ifaces_);
140 }
141
142 const typename FirstIface::ToC &iface() const {
143 return std::get<typename FirstIface::ToC>(ifaces_);
144 }
145
146 private:
147 template<typename T>
148 const typename T::ctype *interface_(const ConfObjectRef &o) const {
149 std::string iface_name = typename T::Info().name();
150 auto iface = static_cast<const typename T::ctype *>(
151 o.get_interface(iface_name));
152 if (!iface && !config_.is_optional(iface_name)) {
153 throw std::runtime_error {
154 "Interface " + iface_name + " not found on " + o.name()
155 };
156 }
157 return iface;
158 }
159
160 template<typename T>
161 void set_ifaces(const ConfObjectRef &o) {
162 std::get<typename T::ToC>(ifaces_) = \
163 typename T::ToC(o.object(), interface_<T>(o));
164 }
165
166 // The variadic template
167 template<typename First, typename... Rest>
168 typename std::enable_if<sizeof...(Rest) != 0, void>::type
169 set_ifaces(const ConfObjectRef &o) {
170 std::get<typename First::ToC>(ifaces_) = \
171 typename First::ToC(o.object(), interface_<First>(o));
172 set_ifaces<Rest...>(o);
173 }
174
175 ifaces_type ifaces_;
176 ConnectConfig config_;
177};
