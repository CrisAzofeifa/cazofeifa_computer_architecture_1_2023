43// @return the next one found starts from the input index
44static size_t find_next(const std::bitset<64> &bits, size_t start_idx) {
45 size_t idx = start_idx;
46 while (idx < bits.size() && !bits.test(idx)) {
47 ++idx;
48 }
49 return idx;
50}
51
52// @return the first one found starts from the lowest bit
53static size_t find_first(const std::bitset<64> &bits) {
54 return find_next(bits, 0);
55}

165 uint64_t get() const override {
166 auto value = val_.to_ullong() & byte_mask_;
167 for (const auto &[offset, f] : fields_) {
168 size_t bits_mask = byte_mask_;
169 if (f->number_of_bits() != number_of_bytes_ * 8) {
170 bits_mask = (1ULL << f->number_of_bits()) - 1;
171 }
172 value &= ~(bits_mask << offset);
173 value |= (f->get() & bits_mask) << offset;
174 }
175 return value;
176 }
177
178 void set(uint64_t value) override {
179 bool changed = val_ != (value & byte_mask_);
180 val_ = value & byte_mask_;
181 for (const auto &[offset, f] : fields_) {
182 size_t bits_mask = byte_mask_;
183 if (f->number_of_bits() != number_of_bytes_ * 8) {
184 bits_mask = (1ULL << f->number_of_bits()) - 1;
185 }
186 f->set((value >> offset) & bits_mask);
187 }
188 if (changed) {
189 SIM_notify(bank_obj_ref(), Sim_Notify_Bank_Register_Value_Change);
190 }
191 }

193 uint64_t read(uint64_t enabled_bits) override {
194 enabled_bits &= byte_mask_;
195 if (enabled_bits == 0) {
196 return 0;
197 }
198
199 std::bitset<64> bits = enabled_bits;
200 unsigned start_bit_offset = find_first(bits);
201 size_t size = bits.count();
202 unsigned end_bit_offset = start_bit_offset + size;
203 if (find_next(bits, end_bit_offset) != 64) {
204 SIM_LOG_ERROR(bank_obj_ref(), 0,
205 "enabled_bits(0x%zx) is malformed: does not contain"
206 " consecutive ones",
207 static_cast<size_t>(enabled_bits));
208 return 0;
209 }
210
211 bool partial = enabled_bits != byte_mask_;
212 uint64_t ret = 0;
213 if (fields_.empty()) {
214 ret = get() & enabled_bits;
215 } else {
216 bits &= val_;
217
218 auto first_field = fields_.lower_bound(start_bit_offset);
219 if (first_field != fields_.begin()) {
220 --first_field;
221 }
222
223 for (auto it = first_field;
224 it != fields_.end() && it->first < end_bit_offset; ++it) {
225 auto[field_offset, field_iface] = *it;
226 size_t field_size = field_iface->number_of_bits();
227
228 // Field has no overlap with the access
229 if (field_offset + field_size <= start_bit_offset) {
230 continue;
231 }
232
233 int bits_to_read = 0;
234 // The lower bits in the field not overlapped with the access
235 unsigned bits_shift = 0;
236 if (field_offset < start_bit_offset) {
237 bits_to_read = std::min<int>(
238 field_offset + field_size - start_bit_offset, size);
239 // for example, read 8 bits from offset 8, the field maps
240 // at [6-10), the bits_mask is 0b1100
241 bits_shift = start_bit_offset - field_offset;
242 } else {
243 bits_to_read = field_offset + field_size - end_bit_offset;
244 if (bits_to_read > 0) {
245 // for example, read 2 bits from offset 0, the field
246 // maps at [0-4), the bit mask is 0b11
247 bits_to_read = field_size - bits_to_read;
248 } else {
249 bits_to_read = field_size;
250 }
251 }
252
253 uint64_t bits_mask = (1ULL << bits_to_read) - 1;
254 bits_mask <<= bits_shift;
255
256 uint64_t field_val = field_iface->read(bits_mask) & bits_mask;
257 field_val >>= bits_shift;
258
259 // The index into the return bits to save the read data
260 unsigned index_of_bits = std::max<int>(field_offset,
261 start_bit_offset);
262 for (auto index = 0; index < bits_to_read; ++index) {
263 bits[index + index_of_bits] = (field_val >> index) & 1;
264 }
265 }
266
267 ret = bits.to_ullong();
268 }
269
270 if (partial) {
271 if (start_bit_offset % 8 == 0 && end_bit_offset % 8 == 0) {
272 SIM_LOG_INFO(4, bank_obj_ref(), 0,
273 "Partial read from register %s: bytes %u-%u ->"
274 " 0x%zx", name().data(), start_bit_offset / 8,
275 end_bit_offset / 8 - 1, static_cast<size_t>(ret));
276 } else {
277 SIM_LOG_INFO(4, bank_obj_ref(), 0,
278 "Partial read from register %s: bits %u-%u ->"
279 " 0x%zx", name().data(), start_bit_offset,
280 end_bit_offset - 1, static_cast<size_t>(ret));
281 }
282 } else {
283 SIM_LOG_INFO(4, bank_obj_ref(), 0,
284 "Read from register %s -> 0x%zx",
285 name().data(), static_cast<size_t>(ret));
286 }
287
288 return ret;
289 }

291 void write(uint64_t value, uint64_t enabled_bits) override {
292 enabled_bits &= byte_mask_;
293 if (enabled_bits == 0) {
294 return;
295 }
296
297 std::bitset<64> bits = enabled_bits;
298 unsigned start_bit_offset = find_first(bits);
299 size_t number_of_bits = bits.count();
300 unsigned end_bit_offset = start_bit_offset + number_of_bits;
301 if (find_next(bits, end_bit_offset) != 64) {
302 SIM_LOG_ERROR(bank_obj_ref(), 0,
303 "enabled_bits(0x%zx) is malformed: does not contain"
304 " consecutive ones",
305 static_cast<size_t>(enabled_bits));
306 return;
307 }
308 if (enabled_bits != byte_mask_) {
309 if (start_bit_offset % 8 == 0 && end_bit_offset % 8 == 0) {
310 SIM_LOG_INFO(4, bank_obj_ref(), 0,
311 "Partial write to register %s: bytes %u-%u"
312 " <- 0x%zx", name().data(), start_bit_offset / 8,
313 end_bit_offset / 8 - 1, value & enabled_bits);
314 } else {
315 SIM_LOG_INFO(4, bank_obj_ref(), 0,
316 "Partial write to register %s: bits %u-%u"
317 " <- 0x%zx", name().data(), start_bit_offset,
318 end_bit_offset - 1, value & enabled_bits);
319 }
320 } else {
321 SIM_LOG_INFO(4, bank_obj_ref(), 0,
322 "Write to register %s <- 0x%zx",
323 name().data(), value & enabled_bits);
324 }
325
326 if (fields_.empty()) {
327 set((get() & ~enabled_bits) | (value & enabled_bits));
328 return;
329 }
330
331 auto first_field = fields_.lower_bound(start_bit_offset);
332 if (first_field != fields_.begin()) {
333 --first_field;
334 }
335
336 for (auto it = first_field;
337 it != fields_.end() && it->first < end_bit_offset; ++it) {
338 auto[field_offset, field_iface] = *it;
339 size_t field_size = field_iface->number_of_bits();
340
341 // Field has no overlap with the access
342 if (field_offset + field_size <= start_bit_offset) {
343 continue;
344 }
345
346 int bits_to_write = 0;
347 // The lower bits in the field not overlapped with the access
348 unsigned bits_shift = 0;
349 if (field_offset < start_bit_offset) {
350 bits_to_write = std::min<int>(
351 field_offset + field_size - start_bit_offset,
352 number_of_bits);
353 // for example, write 8 bits to offset 8, the field maps
354 // at [6-10), the bits_mask is 0b1100
355 bits_shift = start_bit_offset - field_offset;
356 } else {
357 bits_to_write = field_offset + field_size - end_bit_offset;
358 if (bits_to_write > 0) {
359 // for example, read 2 bits from offset 0, the field maps
360 // at [0-4), the bit mask is 0b11
361 bits_to_write = field_size - bits_to_write;
362 } else {
363 bits_to_write = field_size;
364 }
365 }
366
367 uint64_t bits_mask = (1ULL << bits_to_write) - 1;
368 // The index into the param value to fetch the write data
369 unsigned index_of_bits = std::max<int>(field_offset,
370 start_bit_offset);
371 auto write_value = (value >> index_of_bits) & bits_mask;
372 write_value <<= bits_shift;
373 bits_mask <<= bits_shift;
374
375 field_iface->write(write_value, bits_mask);
376 }

379 /*
380 * Parse the field information, add the field or field array into
381 * the resource map
382 */
383 void parse_field(const field_t &f) override {
384 if (dev_obj()->finalized()) {
385 SIM_LOG_ERROR(bank_obj_ref(), 0,
386 "Cannot add fields for register (%s) when device has"
387 " finalized",
388 hierarchical_name().c_str());
389 return;
390 }
391
392 const auto &[name, desc, offset, width] = f;
393
394 if (width == 0) {
395 SIM_LOG_ERROR(bank_obj_ref(), 0,
396 "Ignored invalid field as the width is 0");
397 return;
398 }
399
400 if (name.array_str().empty()) {
401 add_field(name, desc, offset, width);
402 } else {
403 for (const auto &[_name, _offset]
404 : name.arrayNamesToOffsets(width)) {
405 add_field(_name, desc, offset + _offset, width);
406 }
407 }
408 }

451 static void add_register_as_simics_attribute(RegisterInterface *iface) {
452 auto *bank_class = SIM_object_class(iface->bank_obj_ref());
453
454 auto reg_name = iface->name();
455 std::string reg_name_wo_brackets {
456 reg_name.substr(0, reg_name.find('['))
457 };
458
459 if (!SIM_class_has_attribute(bank_class,
460 reg_name_wo_brackets.c_str())) {
461 std::string type {"i"};
462 auto dims = std::count(reg_name.begin(), reg_name.end(), '[');
463 for (int index = 0; index < dims; ++index) {
464 type = '[' + type + "+]";
465 }
466 auto hashed = MapNameToInterfaceObject<
467 RegisterInterface>::hash_str(iface->hierarchical_name());
468 SIM_register_attribute_with_user_data(
469 bank_class, reg_name_wo_brackets.c_str(),
470 get_reg, reinterpret_cast<void *>(hashed),
471 set_reg, reinterpret_cast<void *>(hashed),
472 Sim_Attr_Optional, type.c_str(),
473 iface->description().c_str());
474 }

477 static attr_value_t get_reg_array(int indices, int dim_index,
478 MappableConfObject *obj,
479 const std::string &base_name) {
480 size_t array_index = 0;
481 if (dim_index == indices) {
482 // The most inner dimension
483 std::vector<size_t> values;
484 while (true) {
485 auto *reg_iface = obj->template get_iface<
486 RegisterInterface>(
487 base_name + '[' + std::to_string(array_index) \
488 + ']');
489 if (reg_iface == nullptr) {
490 break;
491 }
492 values.push_back(reg_iface->get());
493 ++array_index;
494 }
495 if (values.empty()) {
496 return SIM_make_attr_nil();
497 } else {
498 return std_to_attr(values);
499 }
500 } else {
501 std::vector<attr_value_t> values;
502 while (true) {
503 auto attr = get_reg_array(indices, dim_index + 1, obj,
504 base_name + '[' \
505 + std::to_string(array_index++) \
506 + ']');
507 if (SIM_attr_is_nil(attr)) {
508 break;
509 } else {
510 values.push_back(attr);
511 }
512 }
513 return std_to_attr(values);
514 }
515 }

517 static attr_value_t get_reg(conf_object_t *obj, void *data) {
518 auto *mappable_obj = from_obj<MappableConfObject>(
519 SIM_port_object_parent(obj));
520 auto name_hash = reinterpret_cast<size_t>(data);
521 auto *reg_iface = mappable_obj->get_iface(name_hash);
522 if (reg_iface == nullptr) {
523 SIM_c_attribute_error("register not found");
524 return SIM_make_attr_nil();
525 }
526
527 auto reg_name = reg_iface->name();
528 auto indices = std::count(reg_name.cbegin(),
529 reg_name.cend(), '[');
530 if (indices == 0) {
531 return std_to_attr(reg_iface->get());
532 }
533
534 // This register contains an array but size unknown
535 auto base_name = std::string(reg_iface->parent()->name()) + SEPARATOR \
536 + std::string(reg_name.substr(0, reg_name.find('[')));
537 return get_reg_array(indices, 1, mappable_obj, base_name);
538 }

540 static set_error_t set_reg_array(int indices, int dim_index,
541 MappableConfObject *obj,
542 const std::string &base_name,
543 attr_value_t *val) {
544 size_t array_index = 0;
545 if (dim_index == indices) {
546 // The most inner dimension
547 auto values = attr_to_std<std::vector<size_t>>(*val);
548 while (true) {
549 auto *reg_iface = obj->template get_iface<
550 RegisterInterface>(
551 base_name + '[' + std::to_string(array_index) \
552 + ']');
553 if (reg_iface == nullptr) {
554 break;
555 } else if (array_index == values.size()) {
556 return Sim_Set_Illegal_Index;
557 }
558 reg_iface->set(values.at(array_index));
559 ++array_index;
560 }
561 if (values.size() == array_index) {
562 return Sim_Set_Ok;
563 } else {
564 return Sim_Set_Illegal_Index;
565 }
566 } else {
567 auto values = attr_to_std<std::vector<attr_value_t>>(*val);
568 while (array_index < values.size()) {
569 auto status = set_reg_array(indices, dim_index + 1, obj,
570 base_name + '[' \
571 + std::to_string(array_index) \
572 + ']', &values.at(array_index));
573 if (status == Sim_Set_Illegal_Index) {
574 if (array_index + 1 == values.size()) {
575 return Sim_Set_Ok;
576 } else {
577 return Sim_Set_Illegal_Index;
578 }
579 }
580 ++array_index;
581 }
582 return Sim_Set_Ok;
583 }
584 }

586 static set_error_t set_reg(conf_object_t *obj, attr_value_t *val,
587 void *data) {
588 auto *mappable_obj = from_obj<MappableConfObject>(
589 SIM_port_object_parent(obj));
590 auto name_hash = reinterpret_cast<size_t>(data);
591 auto reg_iface = mappable_obj->get_iface(name_hash);
592 if (!reg_iface) {
593 return Sim_Set_Interface_Not_Found;
594 }
595
596 auto reg_name = reg_iface->name();
597 auto indices = std::count(reg_name.cbegin(),
598 reg_name.cend(), '[');
599 if (indices == 0) {
600 reg_iface->set(attr_to_std<size_t>(*val));
601 return Sim_Set_Ok;
602 }
603
604 // This register contains an array but size unknown
605 auto base_name = std::string(reg_iface->parent()->name()) + SEPARATOR \
606 + std::string(reg_name.substr(0, reg_name.find('[')));
607 return set_reg_array(indices, 1, mappable_obj, base_name, val);
608 }

610 // Return if the input range overlaps with the existing field range
611 bool has_range_overlap(size_t lsb, size_t msb) const {
612 return std::any_of(fields_.cbegin(), fields_.cend(),
613 [lsb, msb](auto p){
614 return lsb < (p.first + p.second->number_of_bits())
615 && p.first < msb;
616 });
617 }

619 // Add a single field into the resource map
620 void add_field(std::string_view field_name, std::string_view desc,
621 Offset offset, BitWidth width) {
622 if (offset >= number_of_bytes() * 8
623 || offset + width > number_of_bytes() * 8) {
624 SIM_LOG_ERROR(bank_obj_ref(), 0,
625 "Ignored invalid field as the offset exceeds"
626 " the number of bits");
627 return;
628 }
629
630 if (has_range_overlap(offset, offset + width - 1)) {
631 SIM_LOG_ERROR(bank_obj_ref(), 0,
632 "Ignored invalid field as it overlaps with"
633 " other field");
634 return;
635 }
636
637 std::string full_name = hierarchical_name() + SEPARATOR \
638 + field_name.data();
639 fields_[offset] = dev_obj()->get_iface<FieldInterface>(full_name);
640 if (fields_[offset] == nullptr) {
641 allocated_fields_.push_back(
642 std::unique_ptr<FieldInterface>(
643 new Field(dev_obj(), full_name)));
644 fields_[offset] = allocated_fields_.back().get();
645 SIM_LOG_INFO(3, bank_obj_ref(), 0, "Created default field for %s",
646 full_name.c_str());
647 } else if (fields_[offset]->number_of_bits() == 0) {
648 SIM_LOG_INFO(3, bank_obj_ref(), 0, "Used user defined field for %s",
649 full_name.c_str());
650 } else {
651 fields_.erase(offset);
652 SIM_LOG_ERROR(bank_obj_ref(), 0,
653 "Duplicated field name(%s) on same register",
654 field_name.data());
655 return;
656 }
657
658 // Pass the related bits to the field, so it can directly operate
659 // on them
660 bits_type bits;
661 for (size_t i = offset; i < offset + width; ++i) {
662 bits.push_back(val_[i]);
663 }
664 fields_[offset]->init(desc, bits, offset);
665 }
666
667 // The real number of bytes for the register
668 unsigned number_of_bytes_ {8};
669 // The initial value after the register is created
670 uint64_t init_val_ {0};
671 // The register value is represented by val_ & byte_mask_
672 uint64_t byte_mask_ {std::numeric_limits<uint64_t>::max()};
673 // The first bit of the bitset corresponds to the least significant digit
674 // of the number and the last bit corresponds to the most significant digit
675 // i.e. Little-endian bit order
676 std::bitset<64> val_;
677 // The fields mapped on the register. Ordered by the lsb.
678 std::map<size_t, FieldInterface *> fields_;
679 // The parent interface
680 BankInterface *parent_ {nullptr};
681 // @internal: tracks the allocated fields
682 std::vector<std::unique_ptr<FieldInterface>> allocated_fields_;
683};
