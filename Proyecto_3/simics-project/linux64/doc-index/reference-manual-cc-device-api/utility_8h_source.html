31template <typename IFACE>
32class InterfaceMap {
33 template <typename T>
34 using iface_map_t = std::unordered_map<conf_object_t*, T*>;
35
36 public:
37 static iface_map_t<IFACE> &iface_map() {
38 // In C++11, it is thread safe. See §6.7 [stmt.dcl] p4
39 static auto map = std::make_unique<iface_map_t<IFACE>>();
40 return *map.get();
41 }
42};
43
47template <typename IFACE>
48IFACE* get_interface(conf_object_t *obj) {
49 auto &map = InterfaceMap<IFACE>::iface_map();
50 auto it = map.find(obj);
51 if (it != map.end()) {
52 return it->second;
53 }
54
55 auto *iface = dynamic_cast<IFACE*>(from_obj<ConfObject>(obj));
56 if (!iface) {
57 throw std::runtime_error {
58 "The configuration object needs derive from the interface class"
59 };
60 }
61 map[obj] = iface;
62 return iface;
63}
