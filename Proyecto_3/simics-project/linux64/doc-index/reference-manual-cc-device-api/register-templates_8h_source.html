32/*
33 * Register with map information
34 * This class creates a register object and add it to the <arg>bank_iface</arg>.
35 * Optional <arg>fields</arg> is used to create a field object and add it
36 * to the register.
37 * Customized registers such as ReadConstantRegister takes additional arguments.
38 */
39template <typename TRegister = Register, typename... Args>
40class BankRegister : public TRegister {
41 static_assert(std::is_base_of<Register, TRegister>::value,
42 "TRegister must be derived from Register");
43 public:
44 BankRegister(BankInterface *bank_iface,
45 Name reg_name, Description desc, Offset offset, ByteSize size,
46 InitValue value, std::initializer_list<field_t> fields = {},
47 Args... args)
48 : TRegister(bank_iface, reg_name, args ...) {
49 bank_iface->add_register(reg_name, desc, offset, size, value, fields);
50 }
51};

116// Software can only clear bits. This feature is often used when
117// hardware sets bits and software clears them to acknowledge.
118// Software write 1's to clear bits. The new object value is a
119// bitwise AND of the old object value and the bitwise complement
120// of the value written by software.
121class Write1ClearsRegister : public Register {
122 public:
123 using Register::Register;
124
125 void write(uint64_t value, uint64_t enabled_bits) override {
126 Register::write(~value, enabled_bits & value);
127 }
128};

244class OnesRegister : public ConstantRegister {

250 set_description(desc);
251 set_number_of_bytes(number_of_bytes);
252 uint64_t all_ones = std::numeric_limits<uint64_t>::max();
253 if (number_of_bytes != 8) {

259 logged_once_ = true;

266 bool logged_once_ {false};

501// The register is an alias for another register. All operations are forwarded
502// to the other register.
503class AliasRegister : public Register {
504 public:
505 AliasRegister(MappableConfObject *obj, const std::string &name,
506 const std::string &alias_name)
507 : Register(obj, name),
508 alias_name_(alias_name) {
509 init_alias();
510 }
511
512 AliasRegister(BankInterface *parent, std::string_view reg_name,
513 const std::string &alias_name)
514 : Register(parent, reg_name),
515 alias_name_(alias_name) {
516 init_alias();
517 }
518
519 bool is_read_only() const override {
520 return alias_->is_read_only();
521 }
522
523 bool is_mapped() const override {
524 return alias_->is_mapped();
525 }
526
527 uint64_t get() const override {
528 return alias_->get();
529 }
530
531 void set(uint64_t value) override {
532 alias_->set(value);
533 }
534
535 uint64_t read(uint64_t enabled_bits) override {
536 return alias_->read(enabled_bits);
537 }
538
539 void write(uint64_t value, uint64_t enabled_bits) override {
540 alias_->write(value, enabled_bits);
541 }
542
543 std::vector<field_t> fields_info() const override {
544 return alias_->fields_info();
545 }
546
547 private:
548 void init_alias() {
549 if (!HierarchicalObject::is_valid_hierarchical_name(alias_name_)
550 || HierarchicalObject::level_of_hierarchical_name(
551 alias_name_) != static_cast<int>(Level::REGISTER)) {
552 std::string err = "Ignored invalid register name (" \
553 + alias_name_ + ")";
554 SIM_LOG_ERROR(bank_obj_ref(), 0, "%s", err.c_str());
555 throw std::invalid_argument { err };
556 }
557 set_description("Alias register for register " + alias_name_ + ". "
558 + description());
559
560 alias_ = dev_obj()->get_iface<RegisterInterface>(alias_name_);
561 if (!alias_) {
562 // A limitation that depends on the register define order
563 std::string err = "The aliased register " + alias_name_ \
564 + " not found. Alter the register define order to make sure " \
565 + "it is defined before this register.";
566 SIM_LOG_ERROR(bank_obj_ref(), 0, "%s", err.c_str());
567 throw std::invalid_argument { err };
568 }
569 }
570
571 std::string alias_name_;
572 RegisterInterface *alias_ {nullptr};
573};
