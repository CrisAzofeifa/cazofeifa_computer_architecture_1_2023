198 FieldInterface *lookup_field(const std::string &name) const override {
199 FieldInterface *field_interface = nullptr;
200 if (!dev_obj_->finalized()) {
201 SIM_LOG_ERROR(dev_obj_->obj(), 0,
202 "Look up field should be called after finalize"
203 " phase");
204 return field_interface;
205 }
206 if (!is_valid_hierarchical_name(name)) {
207 SIM_LOG_ERROR(dev_obj_->obj(), 0, "Invalid field name: %s",
208 name.c_str());
209 return field_interface;
210 }
211
212 int field_name_level = level_of_hierarchical_name(name);
213 if (field_name_level == 0) {
214 // With field name only like "f2"
215 if (level_ == Level::REGISTER) {
216 field_interface = dev_obj_->get_iface<FieldInterface>(
217 hierarchical_name_ + SEPARATOR + name);
218 } else if (level_ == Level::FIELD) {
219 field_interface = dev_obj_->get_iface<FieldInterface>(
220 hierarchical_name_.substr(0, hierarchical_name_.rfind(
221 SEPARATOR)) \
222 + SEPARATOR + name);
223 } else {
224 SIM_LOG_ERROR(dev_obj_->obj(), 0,
225 "Unable to lookup a field with field name only"
226 " in a bank");
227 }
228 } else if (field_name_level == 1) {
229 // With field name only like "r1.f2"
230 field_interface = dev_obj_->get_iface<FieldInterface>(
231 std::string(bank_name()).append(1, SEPARATOR).append(name));
232 } else {
233 field_interface = dev_obj_->get_iface<FieldInterface>(name);
234 }
235
236 if (!field_interface) {
237 SIM_LOG_ERROR(dev_obj_->obj(), 0, "Lookup field failed: %s",
238 name.c_str());
239 }
240 return field_interface;
241 }

301 static bool is_valid_hierarchical_name(std::string_view name) {
302 if (name.empty()) {
303 return false;
304 }
305 if (isdigit(name[0]) || name[0] == '['
306 || name[0] == ']' || name[0] == '_') {
307 return false;
308 }
309 if (name.find_first_not_of(allowed_characters) != std::string::npos) {
310 return false;
311 }
312 if (std::count(name.cbegin(), name.cend(), SEPARATOR) > 2) {
313 return false;
314 }
315 if (name.find(std::string(2, SEPARATOR)) != std::string::npos) {
316 return false;
317 }
318 if (name.find(SEPARATOR) == 0
319 || name.rfind(SEPARATOR) == name.length() - 1) {
320 return false;
321 }
322 return true;
323 }
324
325 static int level_of_hierarchical_name(std::string_view name) {
326 assert(!name.empty());
327 assert(is_valid_hierarchical_name(name));
328
329 return std::count(name.cbegin(), name.cend(), SEPARATOR);
330 }

332 private:
333 void init() {
334 if (!is_valid_hierarchical_name(hierarchical_name_)) {
335 std::string err = "Cannot set with invalid name string: " \
336 + hierarchical_name_;
337 SIM_LOG_ERROR(dev_obj_->obj(), 0, "%s", err.c_str());
338 throw std::invalid_argument { err };
339 }
340
341 level_ = static_cast<Level>(
342 level_of_hierarchical_name(hierarchical_name_));
343
344 std::string bank_name {hierarchical_name_};
345 if (level_ != Level::BANK) {
346 auto pos_first_separator = hierarchical_name_.find(SEPARATOR);
347 assert(pos_first_separator != std::string::npos);
348 bank_name = hierarchical_name_.substr(0, pos_first_separator);
349 }
350
351 bank_obj_ref_ = SIM_object_descendant(
352 dev_obj_->obj(), ("bank." + bank_name).c_str());
353 if (bank_obj_ref_.object() == nullptr) {
354 std::string err = dev_obj_->obj().name() + ".bank." \
355 + bank_name + " does not exist";
356 SIM_LOG_ERROR(dev_obj_->obj(), 0, "%s", err.c_str());
357 throw std::invalid_argument { err };
358 }
359 }

361 std::string hierarchical_name_;
362 std::string desc_;

364 ConfObjectRef bank_obj_ref_;
365 Level level_;
