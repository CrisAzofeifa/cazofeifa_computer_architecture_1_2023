45// A help struct provides information about a single register
46struct RegisterInfo {
47 RegisterInfo(size_t offset, ByteOrder bo, RegisterInterface *i)
48 : name_(i->name()),
49 desc_(i->description()),
50 size_(i->number_of_bytes()),
51 offset_(offset),
52 bo_(bo),
53 fields_info_(i->fields_info()) {}
54
55 operator attr_value_t() const {
56 attr_value_t info = SIM_alloc_attr_list(6);
57 SIM_attr_list_set_item(&info, 0, SIM_make_attr_string(name_.c_str()));
58 SIM_attr_list_set_item(&info, 1, SIM_make_attr_string(desc_.c_str()));
59 SIM_attr_list_set_item(&info, 2, SIM_make_attr_uint64(size_));
60 SIM_attr_list_set_item(&info, 3, SIM_make_attr_uint64(offset_));
61 attr_value_t fields = SIM_alloc_attr_list(fields_info_.size());
62 unsigned index = 0;
63 for (auto &[name, desc, offset, width] : fields_info_) {
64 attr_value_t field_info = SIM_alloc_attr_list(4);
65 SIM_attr_list_set_item(
66 &field_info, 0,
67 SIM_make_attr_string(name.data()));
68 SIM_attr_list_set_item(
69 &field_info, 1,
70 SIM_make_attr_string(desc.data()));
71 SIM_attr_list_set_item(&field_info, 2,
72 SIM_make_attr_uint64(offset));
73 SIM_attr_list_set_item(&field_info, 3,
74 SIM_make_attr_uint64(offset + width - 1));
75 SIM_attr_list_set_item(&fields, index++, field_info);
76 }
77 SIM_attr_list_set_item(&info, 4, fields);
78 SIM_attr_list_set_item(&info, 5,
79 SIM_make_attr_boolean(bo_ == ByteOrder::BE));
80 return info;
81 }
82
83 private:
84 std::string name_;
85 std::string desc_;
86 size_t size_;
87 size_t offset_;
88 ByteOrder bo_;
89 std::vector<field_t> fields_info_;
90};

94/*
95 * The class for bank port objects
96 *
97 * In Simics, each port object contains exactly one bank, the name of the
98 * bank is determined by the CTOR parameter o. The instance of this class
99 * must have a valid bank interface to direct the received Simics interface
100 * calls.
101 *
102 * @param template parameter TParent must set to the class of the
103 * parent device
104 */
105template <typename TParent>
106class BankPort : public Port<TParent>,
107 public BankPortInterface,
108 public BankInstrumentationSubscribeConnection,
109 public iface::RegisterViewInterface,
110 public iface::RegisterViewReadOnlyInterface,
111 public iface::TransactionInterface {
112 static_assert(
113 std::is_base_of<MappableConfObject, TParent>::value,
114 "BankPort requires the parent class be a MappableConfObject");
115
116 public:
117 /*
118 * Used by the "by code" option to set a bank on the port
119 * By this option, the bank is created as a class member variable of
120 * the user defined bank port class. Thus no need to initialize the
121 * bank_iface here.
122 */
123 explicit BankPort(ConfObjectRef o)
124 : Port<TParent>(o),
125 bank_name_(bank_name_from_port_obj(o)) {
126 dev_obj()->set_bank_as_initialized(bank_name_);
127 SIM_hap_add_callback_obj(
128 "Core_Conf_Object_Created", o.object(), 0,
129 reinterpret_cast<obj_hap_func_t>(object_created), nullptr);
130 }
131
132 /*
133 * Used by the "by data" option to set a bank on the port
134 * By this option, bank is created and initialized here.
135 */
136 BankPort(ConfObjectRef o, const bank_t *bank)
137 : BankPort(o) {
138 if (bank == nullptr) {
139 throw std::invalid_argument("Bank pointer cannot be nullptr");
140 }
141 set_bank(*bank);
142 }
143
144 virtual ~BankPort() {
145 SIM_hap_delete_callback_obj(
146 "Core_Conf_Object_Created", Port<TParent>::obj().object(),
147 reinterpret_cast<obj_hap_func_t>(object_created), nullptr);
148 }
149
150 // BankPortInterface
151 std::string_view bank_name() const override {
152 return bank_name_;
153 }
154
155 const BankInterface *bank_iface() const override {
156 return bank_iface_;
157 }
158
159 MappableConfObject *dev_obj() const override {
160 return Port<TParent>::parent();
161 };
162
163 bool validate_bank_iface() const override {
164#if __cplusplus >= 202002L || (defined(_MSVC_LANG) && _MSVC_LANG >= 202002L)
165 if (bank_iface_ == nullptr) { [[unlikely]]
166#else
167 if (bank_iface_ == nullptr) {
168#endif
169 SIM_LOG_ERROR(Port<TParent>::obj(), 0,
170 "BankPort should have one bank");
171 return false;
172 }
173 return true;
174 }
175
176 void set_bank(const bank_t &bank) override {
177 if (bank_iface_) {
178 SIM_LOG_ERROR(Port<TParent>::obj(), 0,
179 "bank iface can only be set once");
180 return;
181 }
182
183 bank_iface_ = dev_obj()->template get_iface<BankInterface>(bank_name_);
184 if (bank_iface_) {
185 SIM_LOG_INFO(3, Port<TParent>::obj(), 0,
186 "Used user defined bank for %s",
187 bank_name_.c_str());
188 } else {
189 allocated_bank_.reset(new Bank(dev_obj(), bank_name_));
190 bank_iface_ = allocated_bank_.get();
191 SIM_LOG_INFO(3, Port<TParent>::obj(), 0,
192 "Created a new default bank for %s",
193 bank_name_.c_str());
194 }
195 auto &[name, desc, registers] = bank;
196 bank_iface_->set_description(desc);
197 bank_iface_->set_callbacks(this);
198 for (auto &reg : registers) {
199 bank_iface_->add_register(reg);
200 }
201 }
202
203 // iface::TransactionInterface
204 exception_type_t issue(transaction_t *t, uint64 addr) override {
205 if (!validate_bank_iface()) {
206 return Sim_PE_IO_Not_Taken;
207 }
208 return bank_iface_->transaction_access(t, addr);
209 }
210
211 // iface::RegisterViewInterface
212 const char *description() override {
213 if (!validate_bank_iface()) {
214 return nullptr;
215 }
216 return bank_iface_->description().data();
217 }
218 bool big_endian_bitorder() override {
219 return Port<TParent>::parent()->big_endian_bitorder();
220 }
221 unsigned number_of_registers() override {
222 if (!validate_bank_iface()) {
223 return 0;
224 }
225 return bank_iface_->number_of_registers();
226 }
227 attr_value_t register_info(unsigned reg) override {
228 if (!validate_bank_iface()) {
229 return SIM_make_attr_nil();
230 }
231 auto[reg_offset, reg_iface] = bank_iface_->register_at_index(reg);
232 if (reg_iface) {
233 return detail::RegisterInfo(reg_offset,
234 bank_iface_->get_byte_order(),
235 reg_iface);
236 }
237 return SIM_make_attr_nil();
238 }
239 uint64 get_register_value(unsigned reg) override {
240 if (!validate_bank_iface()) {
241 return 0;
242 }
243 auto[reg_offset, reg_iface] = bank_iface_->register_at_index(reg);
244 return reg_iface ? reg_iface->get() : 0;
245 }
246 void set_register_value(unsigned reg, uint64 val) override {
247 if (validate_bank_iface()) {
248 auto[reg_offset, reg_iface] = bank_iface_->register_at_index(reg);
249 if (reg_iface) {
250 reg_iface->set(val);
251 }
252 }
253 }
254
255 // iface::RegisterViewReadOnlyInterface
256 bool is_read_only(unsigned reg) override {
257 if (!validate_bank_iface()) {
258 return false;
259 }
260 auto[reg_offset, reg_iface] = bank_iface_->register_at_index(reg);
261 return reg_iface ? reg_iface->is_read_only() : false;
262 }
263
264 private:
265 // Function get called when the object is created and finalized
266 static void object_created(lang_void *, conf_object_t *obj) {
267 get_interface<BankPortInterface>(obj)->validate_bank_iface();
268 }
269
270 static std::string_view bank_name_from_port_obj(ConfObjectRef port_obj) {
271 std::string_view port_name = port_obj.name();
272 if (port_name.find(bank_name_keyword) == std::string::npos) {
273 throw std::invalid_argument(
274 "Invalid bank port name (" + port_obj.name() + ")");
275 }
276 return port_name.substr(port_name.rfind(bank_name_keyword) + \
277 bank_name_keyword.size());
278 }
279
280 // A keyword exists in the name of all bank ports
281 static constexpr std::string_view bank_name_keyword {".bank."};
282 // Points to the actual bank interface
283 BankInterface *bank_iface_ { nullptr };
284 // Name of the bank
285 std::string bank_name_;
286 // @internal: record the heap allocated Bank
287 std::unique_ptr<BankInterface> allocated_bank_;
288};

290template <typename TBankPort> ConfClassPtr
291make_bank_port(const std::string &name, const std::string &desc) {
292 auto port = simics::make_class<TBankPort>(name, "", desc);
293 port->add(simics::iface::TransactionInterface::Info());
294 port->add(simics::iface::RegisterViewInterface::Info());
295 port->add(simics::iface::RegisterViewReadOnlyInterface::Info());
296 port->add(simics::iface::BankInstrumentationSubscribeInterface::Info());
297 port->add(simics::iface::InstrumentationOrderInterface::Info());
298 SIM_register_notifier(*port, Sim_Notify_Bank_Register_Value_Change,
299 "triggered upon register value changes");
300 return port;
301}

304make_bank_port(const std::string &name, const std::string &desc, TArg *arg) {
305 auto port = simics::make_class<TBankPort>(name, "", desc, arg);
306 port->add(simics::iface::TransactionInterface::Info());
307 port->add(simics::iface::RegisterViewInterface::Info());
308 port->add(simics::iface::RegisterViewReadOnlyInterface::Info());
309 port->add(simics::iface::BankInstrumentationSubscribeInterface::Info());
310 port->add(simics::iface::InstrumentationOrderInterface::Info());
311 SIM_register_notifier(*port, Sim_Notify_Bank_Register_Value_Change,
312 "triggered upon register value changes");
313 return port;
314}

316/*
317 * Register the bank as port object, pass the pointer to the registers as
318 * the class data.
319 */
320template <typename TParent>
321void create_hierarchy_from_register_data(ConfClass *cls, const bank_t &bank) {
322 const auto &[name, desc, registers] = bank;
323 cls->add(make_bank_port<BankPort<TParent>, const bank_t>(
324 cls->name() + SEPARATOR + std::string(name.base_name()),
325 std::string(desc.data()), &bank),
326 std::string("bank.") + name.data());
327}
328
329template <typename TParent>
330void create_hierarchy_from_register_data(
331 ConfClass *cls, std::initializer_list<bank_t> register_data) {
332 for (auto &bank : register_data) {
333 create_hierarchy_from_register_data<TParent>(cls, bank);
334 }
335}
