28class PortBank : public TBank {
29 static_assert(std::is_base_of<Bank, TBank>::value,
30 "TBank must be derived from Bank");
31 public:
32 PortBank(BankPortInterface *port_iface, Description desc, Args... args)
33 : TBank(port_iface->dev_obj(), port_iface->bank_name().data(),
34 args ...) {

36 }
37};
