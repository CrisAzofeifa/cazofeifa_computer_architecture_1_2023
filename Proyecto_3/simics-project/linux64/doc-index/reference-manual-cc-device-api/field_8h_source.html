33/*
34 * A field with default behavior
35 *
36 * A field consists of several or all bits from a register. The default
37 * behavior is all bits support both read and write without any side effects.
38 */
39class Field : public HierarchicalObject,
40 public FieldInterface {
41 public:
42 // Construct from hierarchical name
43 // @param name begins with the bank name, e.g, "bankA.registerB.fieldC"
44 Field(MappableConfObject *dev_obj, const std::string &name)
45 : HierarchicalObject(dev_obj, name, this) {}
46
47 // Construct from parent
48 Field(RegisterInterface *parent, std::string_view field_name)
49 : HierarchicalObject(
50 parent->dev_obj(),
51 parent->hierarchical_name() + SEPARATOR + field_name.data(),
52 this),
53 parent_(parent) {}
54
55 // No duplication
56 Field(const Field&) = delete;
57 Field& operator=(const Field&) = delete;
58
59 Field(Field &&rhs)
60 : HierarchicalObject(std::move(rhs)),
61 bits_(std::move(rhs.bits_)) {
62 HierarchicalObject::init_iface(this);
63 }
64 Field& operator=(Field&& rhs) {
65 // check for self-assignment
66 if (this == &rhs)
67 return *this;
68
69 HierarchicalObject::operator=(std::move(rhs));
70 HierarchicalObject::init_iface(this);
71 bits_ = std::move(rhs.bits_);
72 return *this;
73 }
74
75 // FieldInterface
76 std::string_view name() const override {
77 return HierarchicalObject::name();
78 }
79
80 const std::string &description() const override {
81 return HierarchicalObject::description();
82 }
83
84 unsigned number_of_bits() const override {
85 return bits_.size();
86 }
87
88 void init(std::string_view desc, const bits_type &bits,
89 int8_t offset) override {
90 if (number_of_bits() != 0) {
91 SIM_LOG_ERROR(bank_obj_ref(), 0,
92 "Re-init field (%s) is not allowed",
93 hierarchical_name().c_str());
94 return;
95 }
96 set_description(desc);
97 set_bits(bits);
98 offset_ = offset;
99 if (parent_ == nullptr) {
100 parent_ = dev_obj()->get_iface<RegisterInterface>(
101 std::string(parent_name()));
102 }
103 }
104
105 RegisterInterface *parent() const override {
106 return parent_;
107 }
108
109 uint64_t get() const override {
110 // Avoid unwanted casting of the result (int to uint64_t)
111 uint64_t init_val = 0;
112 return std::accumulate(bits_.rbegin(), bits_.rend(), init_val,
113 [](uint64_t v, bool e) { return (v << 1) + e; });
114 }
115
116 void set(uint64_t value) override {
117 if (get() != value) {
118 for (auto &bit : bits_) {
119 bit = value & 1;
120 value >>= 1;
121 }
122 SIM_notify(bank_obj_ref(), Sim_Notify_Bank_Register_Value_Change);
123 }
124 }
125
126 uint64_t read(uint64_t enabled_bits) override {
127 if (enabled_bits == 0) {
128 return 0;
129 }
130 return get() & enabled_bits;
131 }
132
133 void write(uint64_t value, uint64_t enabled_bits) override {
134 if (enabled_bits != 0) {
135 set((get() & ~enabled_bits) | (value & enabled_bits));
136 }
137 }
138
139 size_t offset() const {
140 return offset_;
141 }
142
143 // Set the bits for this field. They are references to the bits
144 // in the corresponding register
145 void set_bits(const bits_type &bits) {
146 if (dev_obj()->finalized()) {
147 SIM_LOG_ERROR(bank_obj_ref(), 0,
148 "Cannot set bits for field (%s) when device has"
149 " finalized",
150 hierarchical_name().c_str());
151 return;
152 }
153 bits_ = bits;
154 }
155
156 private:
157 // The first bit (bits_[0]) corresponds to the least significant digit of
158 // the value and the last bit corresponds to the most significant digit
159 bits_type bits_;
160 // The offset of the first bit in containing register
161 std::int8_t offset_ {-1};
162
163 // The parent interface
164 RegisterInterface *parent_ {nullptr};
165};

167} // namespace simics
