35// Class supports get/set a pointer to IFACE with a string name
36template <typename IFACE>
37class MapNameToInterfaceObject : public MapNameToInterface<IFACE> {
38 public:
39 static size_t hash_str(const std::string &name) {
40 return std::hash<std::string>{}(name);
41 }
42
43 // MapNameToInterface<IFACE>
44 void set_iface(const std::string &name, IFACE *iface) override {
45 if (!iface) {
46 throw std::invalid_argument {
47 "Cannot set with NULL interface"
48 };
49 } else if (name.empty()) {
50 throw std::invalid_argument {
51 "Cannot set with empty name string"
52 };
53 } else {
54 name_to_iface_[hash_str(name)] = iface;
55 }
56 }
57 IFACE *get_iface(const std::string &name) const override {
58 auto hashed = hash_str(name);
59 if (name_to_iface_.find(hashed) == name_to_iface_.end()) {
60 return nullptr;
61 }
62 return name_to_iface_.at(hashed);
63 }
64 void erase_iface(const std::string &name) override {
65 name_to_iface_.erase(hash_str(name));
66 }
67
68 private:
69 std::unordered_map<size_t, IFACE *> name_to_iface_;
70};

122/*
123 * A class inherited by a model class to support memory mapped I/O
124 *
125 * It manages a container which associates a unique name to its corresponding
126 * operational interface. These 3 object types are supported: bank, register
127 * and field.
128 */
129class MappableConfObject : public ConfObject {
130 public:
131 explicit MappableConfObject(ConfObjectRef obj) : ConfObject(obj) {}
132 virtual ~MappableConfObject() {}
133
134 // Set the object interface by a name.
135 template <typename IFACE>
136 void set_iface(const std::string &name, IFACE *iface) {
137 if (finalized()) {
138 SIM_LOG_ERROR(obj(), 0,
139 "Cannot set interface for %s when ConfObject"
140 " has been finalized", name.c_str());
141 return;
142 }
143
144 auto current_iface = get_iface<IFACE>(name);
145 if (current_iface && current_iface != iface) {
146 if (is_bank_initialized(name)) {
147 SIM_LOG_INFO(4, obj(), 0,
148 "Interface for %s ignored since bank"
149 " has been initialized", name.c_str());
150 return;
151 }
152 // TODO(xiuliang): Logs when interface is overridden
153 }
154
155 try {
156 std::get<MapNameToInterfaceObject<IFACE>>(
157 objects_).set_iface(name, iface);
158 } catch (std::exception &e) {
159 SIM_LOG_ERROR(obj(), 0, "%s", e.what());
160 }
161 }
162
163 /* Get the object interface by a name.
164 *
165 * User should check if the return pointer is nullptr before use.
166 */
167 template <typename IFACE>

196 std::vector<size_t> register_dimensions(const std::string &name) {
197 auto reg_name = name.substr(name.rfind('.') + 1);
198 auto num_dims = std::count(reg_name.cbegin(), reg_name.cend(), '[');
199 if (num_dims == 0) {
200 return {};
201 }
202 std::vector<size_t> dims(num_dims);
203
204 auto all_regs = all_registers();
205 auto base_name = name.substr(0, name.rfind('.') + 1) + \
206 reg_name.substr(0, reg_name.find('['));
207 for (int dim_index = 0; dim_index < num_dims; ++dim_index) {
208 auto prefix = base_name;
209 for (int prefix_index = 0; prefix_index < dim_index;
210 ++prefix_index) {
211 prefix.append("[0]");
212 }
213 std::string postfix;
214 for (int postfix_index = dim_index + 1; postfix_index < num_dims;
215 ++postfix_index) {
216 postfix.append("[0]");
217 }
218 size_t s = 0;
219
220 while (true) {
221 std::string check_name = prefix + '[' + std::to_string(s) + \
222 ']' + postfix;
223 if (all_regs.find(check_name) != all_regs.end()) {
224 ++s;
225 } else {
226 break;
227 }
228 }
229 dims[dim_index] = s;
230 }
231 return dims;
232 }

234 void set_bank_as_initialized(const std::string &name) {
235 initialized_banks_.insert(name);
236 }
237
238 private:
239 bool is_bank_initialized(const std::string &name) {
240 const auto &bank_name = name.substr(0, name.find('.'));
241 return initialized_banks_.find(bank_name) != initialized_banks_.end();
242 }

247 MapNameToInterfaceObject<FieldInterface>> objects_;
248
249 // The name of initialized banks
250 std::unordered_set<std::string> initialized_banks_;
251};
