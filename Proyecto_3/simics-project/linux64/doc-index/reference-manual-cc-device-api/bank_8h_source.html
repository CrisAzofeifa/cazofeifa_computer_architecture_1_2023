43/*
44 * A bank with default behavior
45 *
46 * The default behavior is to allow access to any offset, without any
47 * side-effects. Registers and fields don't have to be mapped as all reads
48 * return zero and all writes are ignored.
49 */
50class Bank : public BankInterface,
51 public HierarchicalObject {
52 public:
53 // @param name is the bank name alone, e.g., b0
54 Bank(MappableConfObject *dev_obj, const std::string &name)
55 : HierarchicalObject(dev_obj, name, this) {}
56
57 // No duplication
58 Bank(const Bank&) = delete;
59 Bank& operator=(const Bank&) = delete;
60
61 Bank(Bank &&rhs)
62 : HierarchicalObject(std::move(rhs)),
63 byte_order_(rhs.byte_order_),
64 regs_(std::move(rhs.regs_)),
65 newd_regs_(std::move(rhs.newd_regs_)),
66 callbacks_(rhs.callbacks_) {
67 HierarchicalObject::init_iface(this);
68 rhs.callbacks_ = nullptr;
69 }
70 Bank& operator=(Bank&& rhs) {
71 // check for self-assignment
72 if (this == &rhs)
73 return *this;
74
75 HierarchicalObject::operator=(std::move(rhs));
76 HierarchicalObject::init_iface(this);
77 byte_order_ = rhs.byte_order_;
78 regs_ = std::move(rhs.regs_);
79 newd_regs_ = std::move(rhs.newd_regs_);
80 callbacks_ = rhs.callbacks_;
81 rhs.callbacks_ = nullptr;
82 return *this;
83 }
