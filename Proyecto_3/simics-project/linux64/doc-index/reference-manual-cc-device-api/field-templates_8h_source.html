28/*
29 * Field with map information
30 * This class creates a field object and add it to the <arg>reg_iface</arg>.
31 * Customized fields such as ReadConstantField takes additional arguments.
32 */
33template <typename TField = Field, typename... Args>
34class RegisterField : public TField {
35 static_assert(std::is_base_of<Field, TField>::value,
36 "TField must be derived from Field");
37 public:
38 RegisterField(RegisterInterface *reg_iface, Name name, Description desc,
39 Offset offset, BitWidth size, Args... args)
40 : TField(reg_iface, name, args ...) {
41 reg_iface->parse_field({name, desc, offset, size});
42 }
43};

45// Templates
46// Writes are ignored.
47class IgnoreWriteField : public Field {
48 public:
49 using Field::Field;
50
51 void write(uint64_t value, uint64_t enabled_bits) override {}

54// Reads return 0. Writes are unaffected by this template.
55class Read0Field : public Field {
56 public:
57 using Field::Field;
58
59 uint64_t read(uint64_t enabled_bits) override {
60 SIM_LOG_INFO(4, bank_obj_ref(), 0,
61 "Read from read-zero field %s -> 0x0.",
62 name().data());
63 return 0;
64 }

67// The object value is read-only for software, the object value
68// can be modified by hardware.
69class ReadOnlyField : public Field {
70 public:
71 using Field::Field;
72
73 void write(uint64_t value, uint64_t enabled_bits) override {
74 if ((value & enabled_bits) != (get() & enabled_bits)) {
75 SIM_LOG_SPEC_VIOLATION(logged_once_ ? 2 : 1, bank_obj_ref(), 0,
76 "Write to read-only field %s (value written"
77 " = 0x%08zx, contents = 0x%08zx).",
78 name().data(), value & enabled_bits, get());
79 logged_once_ = true;
80 }
81 }
82
83 private:
84 bool logged_once_ {false};
85};

87// Software can only clear bits. This feature is often used when
88// hardware sets bits and software clears them to acknowledge.
89// Software write 1's to clear bits. The new object value is a
90// bitwise AND of the old object value and the bitwise complement
91// of the value written by software.
92class Write1ClearsField : public Field {
93 public:
94 using Field::Field;
95
96 void write(uint64_t value, uint64_t enabled_bits) override {
97 Field::write(~value, enabled_bits & value);
98 }
99};

101// Software reads return the object value. The object value is
102// then reset to 0 as a side-effect of the read.
103class ClearOnReadField : public Field {
104 public:
105 using Field::Field;
106
107 uint64_t read(uint64_t enabled_bits) override {
108 uint64_t value = get();
109 set(0);
110 return value & enabled_bits;
111 }

114// Software can only set bits to 1. The new object value is

159// Writes are forbidden and have no effect. TODO(xiuliang): no_reset?
160class ConstantField : public Field {
161 public:
162 ConstantField(MappableConfObject *obj, const std::string &name,
163 uint64_t init_val) :
164 Field(obj, name),
165 init_val_(init_val) {}
166
167 ConstantField(RegisterInterface *parent, std::string_view field_name,
168 uint64_t init_val) :
169 Field(parent, field_name),
170 init_val_(init_val) {}
171
172 void write(uint64_t value, uint64_t enabled_bits) override {
173 if ((value & enabled_bits) != (get() & enabled_bits)) {
174 SIM_LOG_SPEC_VIOLATION(logged_once_ ? 2 : 1, bank_obj_ref(), 0,
175 "Write to constant field %s"

177 " contents = 0x%08zx).", name().data(),
178 value & enabled_bits, get());
179 logged_once_ = true;
180 }
181 }
182
183 void init(std::string_view desc, const bits_type &bits,
184 int8_t offset) override {
185 Field::init(desc, bits, offset);
186 set(init_val_);
187 }
188
189 private:
190 bool logged_once_ {false};
191 uint64_t init_val_;
192};

406// The object's functionality is not in the model's scope and has been
407// left unimplemented as a design decision. Software and hardware writes
408// and reads are implemented as default writes and reads. Debug fields
409// are a prime example of when to use this template. This is different from
410// unimplemented which is intended to be implement (if required) but is a
411// limitation in the current model.
412class DesignLimitationField : public Field {
413 public:
414 DesignLimitationField(MappableConfObject *obj, const std::string &name)
415 : Field(obj, name) {
416 set_description(std::string("Not implemented (design limitation).")
417 + " This field is a dummy field with no side effects. "
418 + description());
419 }
420
421 DesignLimitationField(RegisterInterface *parent,
422 std::string_view field_name)
423 : Field(parent, field_name) {
424 set_description(std::string("Not implemented (design limitation).")
425 + " This field is a dummy field with no side effects. "
426 + description());
