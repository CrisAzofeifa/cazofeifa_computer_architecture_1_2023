39 protected:
40 static int send_message(conf_object_t *obj, conf_object_t *src,
41 pcie_message_type_t type, byte_string_t payload) {
42 // Need to convert between Simics byte_string_t and C++ type
43 std::vector<uint8_t> p(payload.str, payload.str+payload.len);
44 return adapter<TBase, TInterface>(obj)->send_message(type, p);
45 }
46
47 private:
48 std::vector<std::string> description(conf_object_t *obj,
49 DescriptionType type) {
50 return descriptionBase<TBase, TInterface>(obj, type);
51 }
52 pci_express_interface_t init_iface() {
53 pci_express_interface_t iface = {};
54 iface.send_message = send_message;
55 return iface;
56 }
