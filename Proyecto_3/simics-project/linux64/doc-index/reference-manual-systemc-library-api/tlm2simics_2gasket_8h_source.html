116 private:
117 void b_transport(tlm::tlm_generic_payload &trans, // NOLINT
118 sc_core::sc_time &local_time_offset) { // NOLINT
119 if (trans.get_byte_enable_ptr() != NULL &&
120 !transaction_handler_->byte_enable_supported(simics_obj_, &trans)) {
121 return;
122 }
123
124 // Before we send the transaction to simics we need to synchronize
125 // local time.
126 if (local_time_offset.value() > 0) {
127#ifndef RISC_SIMICS
128 wait(local_time_offset);
129#endif
130 local_time_offset = sc_core::SC_ZERO_TIME;
131 }
132
133 ProcessStackHandler pstack(internal());
134 trans.set_response_status(tlm::TLM_OK_RESPONSE);
135 tlm::tlm_response_status status = tlm::TLM_INCOMPLETE_RESPONSE;
136 {
137 // Switching context to detect re-entry without setting context.
138 Context c(&null_simulation_);
139 status = transaction_handler_->simics_transaction(simics_obj_,
140 &trans);
141 }
142 if (trans.get_response_status() != status &&
143 status != tlm::TLM_OK_RESPONSE) {
144 const char *msg = "Gasket transaction error";
145 switch (status) {
146 case tlm::TLM_INCOMPLETE_RESPONSE: {
147 msg = "Gasket did not attempt to execute the command";
148 break;
149 }
150 case tlm::TLM_ADDRESS_ERROR_RESPONSE: {
151 msg = "Gasket was unable to act on the address attribute, "
152 "or address out-of-range";
153 break;
154 }
155 case tlm::TLM_COMMAND_ERROR_RESPONSE: {
156 msg = "Gasket was unable to execute the command";
157 break;
158 }
159 case tlm::TLM_BURST_ERROR_RESPONSE: {
160 msg = "Gasket was unable to act on the data length or "
161 "streaming width";
162 break;
163 }
164 case tlm::TLM_BYTE_ENABLE_ERROR_RESPONSE: {
165 msg = "Gasket was unable to act on the byte enable";
166 break;
167 }
168 case tlm::TLM_GENERIC_ERROR_RESPONSE: {
169 break;
170 }
171 default: {
172 }
173 }
174 SIM_LOG_SPEC_VIOLATION(1, simics_obj_, 0, "%s", msg);
175 }
176
177 trans.set_response_status(status);
178 transaction_handler_->update_dmi_allowed(simics_obj_, &trans);
179 }
180
181 unsigned int transport_dbg(tlm::tlm_generic_payload &trans) { // NOLINT
182 ProcessStackHandler pstack(internal());
183 return transaction_handler_->debug_transaction(simics_obj_, &trans);
184 }
185
186 bool get_direct_mem_ptr(tlm::tlm_generic_payload &trans, // NOLINT
187 tlm::tlm_dmi &dmi_data) { // NOLINT
188 ProcessStackHandler pstack(internal());
189 return transaction_handler_->get_direct_mem_ptr(simics_obj_,
190 trans, dmi_data);
191 }

193 InternalInterface *internal() {
194 if (!internal_) {
195 Adapter *adapter = static_cast<Adapter *>(
196 SIM_object_data(simics_obj_));
197 assert(adapter);
198 internal_ = adapter->internal();
199 }
200
201 return internal_;
202 }

211 sc_core::sc_object *initiator_socket_;
212};
