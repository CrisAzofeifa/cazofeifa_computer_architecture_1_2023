161 bool addrSpaceValue(AttrDictParser *parser, simics::types::addr_space_t *v,

166
167 if (space > types::Sim_Addr_Space_Memory) {
168 parser->reportError("%s must be in range [%i - %i]",
169 key.c_str(),
170 types::Sim_Addr_Space_Conf,
171 types::Sim_Addr_Space_Memory);
172 return false;
173 }
174
175 *v = static_cast<types::addr_space_t>(space);
176 return true;
