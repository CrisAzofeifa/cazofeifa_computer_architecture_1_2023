31class SerialDeviceExtension : public Extension<SerialDeviceExtension,
32 SerialDeviceInterface> {
33 public:
34 virtual void call(SerialDeviceInterface *device) {
35 switch (method_.value<Method>()) {
36 case WRITE:
37 method_return_ = device->write(method_input_[0].value<int>());
38 break;
39 case RECEIVE_READY:
40 device->receive_ready();
41 break;
42 }
43 }
44
45 virtual int write(int value) {
46 method_input_.push_back(value);
47 method_ = WRITE;
48 send();
49 return method_return_.value<int>();
50 }
51 virtual void receive_ready() {
52 method_ = RECEIVE_READY;
53 send();
54 }
55
56 private:
57 enum Method {
58 WRITE,
59 RECEIVE_READY
60 };
61};
