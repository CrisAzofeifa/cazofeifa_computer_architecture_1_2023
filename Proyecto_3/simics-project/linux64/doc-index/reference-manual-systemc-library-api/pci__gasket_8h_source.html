79template<unsigned int BUSWIDTH = 32,
80 typename TYPES = tlm::tlm_base_protocol_types>
81class PciGasket : public PciGasketBase,
82 public simics2tlm::IoMemoryGasketAdapter,
83 public simics2tlm::PciDeviceGasketAdapter,
84 public simics2tlm::PciExpressGasketAdapter {
85 public:
86 explicit PciGasket(iface::SimulationInterface *simulation)
87 : IoMemoryGasketAdapter(&systemc_mmio_, simulation),
88 PciDeviceGasketAdapter(&systemc_pci_, simulation),
89 PciExpressGasketAdapter(&systemc_pci_express_, simulation),
90 simulation_(simulation) {
91 }
92
93 // PciDeviceGasketAdapter
94 virtual void bus_reset() { /* override */
95 Context context(simulation_);
96 interconnect_.busReset();
97 PciDeviceGasketAdapter::bus_reset();
98 }
99
100 // NOTE: the template parameter is required for the
101 // PciMappingInterconnect::connect*<> template methods, where it is
102 // needed for the multi_passthrough_target_socket<> template class. By
103 // using the TPciDevice as method parameter, the compiler can deduce the
104 // template parameter.
105 template<typename TPciDevice>
106 void connect(TPciDevice *device) {
107 connect(dynamic_cast<iface::PciDeviceQueryInterface *>(device),
108 dynamic_cast<iface::BaseAddressRegisterQueryInterface *>(
109 device),
110 simulation_->simics_object());
111 }
112
113 void connect(iface::PciDeviceQueryInterface *pci,
114 iface::BaseAddressRegisterQueryInterface *bar,
115 ConfObjectRef gasket) {
116 ConfObjectRef o = simulation_->simics_object();
117
118 if (!pci) {
119 SIM_LOG_ERROR(gasket, Log_Configuration,
120 "The PCI device does not implement the"
121 " PciDeviceQueryInterface interface");
122 return;
123 }
124
125 if (!bar) {
126 SIM_LOG_ERROR(gasket, Log_Configuration,
127 "The PCI device does not implement the"
128 " BaseAddressRegisterQueryInterface interface");
129 return;
130 }
131
132 // Connect the device's config space target socket and pci-bus
133 // initiator socket via the interconnect to support snooping
134 interconnect_.connect(pci, o);
135
136 // Connecting the interconnect to the pci-bus
137 simics_pci_->set_device(gasket);
138 simics_pci_->set_gasket(
139 tlm2simics::createGasket(
140 &interconnect_.pci_bus_initiator_socket, o));
141
142 // Connect device's pci-device target socket
143 // The bus_reset() is intercepted by the composite class and handled
144 // by busReset() before the transaction is forwarded to the device.
145 // This allows the IC to unmap all mappings upon bus_reset.
146 systemc_pci_.set_gasket(
147 simics2tlm::createGasket(
148 &interconnect_.simics_target_socket, o));
149 systemc_pci_express_.set_gasket(
150 simics2tlm::createGasket(
151 &interconnect_.simics_target_socket, o));
152
153 // Connect IC's config space target sockets (snooping)
154 interconnect_.connectConfig(bar, &systemc_mmio_, o);
155
156 // Connect device's io/mem space target socket(s)
157 interconnect_.connectMmio(bar, &systemc_mmio_, o);
158 }
159
160 private:
161 iface::SimulationInterface *simulation_;
162
163 // Snoop PCI transactions and handle Simics mappings
164 PciMappingInterconnect<BUSWIDTH, TYPES> interconnect_;
165
166 // Gaskets required for wrapping a PCI device
167 simics2tlm::IoMemory systemc_mmio_; // incoming MMIO (conf/mem/io spaces)
168 simics2tlm::PciDevice systemc_pci_; // Simics pci-bus specific interface
169 simics2tlm::PciExpress systemc_pci_express_; // Simics PCIe interface
170};
