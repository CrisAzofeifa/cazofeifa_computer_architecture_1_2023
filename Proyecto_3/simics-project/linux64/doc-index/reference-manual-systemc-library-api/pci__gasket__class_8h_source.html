35template<unsigned int BUSWIDTH = 32,
36 typename TYPES = tlm::tlm_base_protocol_types>
37class PciGasketClass
38 : public ConfObject, public PciGasket<BUSWIDTH, TYPES>,
39 public GasketClassInterface, public Version,
40 public SimulationInterfaceProxy {
41 public:
42 explicit PciGasketClass(ConfObjectRef o)
43 : ConfObject(o), PciGasket<BUSWIDTH, TYPES>(this) {}
44 virtual void finalize() {
45 // Avoid recursive calls to finalize_instance
46 SIM_set_object_configured(obj());
47
48 // Make sure the Adapter is configured
49 simulation_ref_.require();
50 }

52 // GasketClassInterface
53 virtual iface::ScVersionInterface *version() {
54 return this;
55 }
56 virtual void createGasket(iface::SimulationInterface *simulation) {
57 setSimulation(simulation);
58 sc_core::sc_object *obj = sc_core::sc_find_object(
59 sc_pci_device_name_.c_str());
60 if (!obj) {
61 SIM_LOG_ERROR(simulation->simics_object(), Log_Configuration,
62 "Invalid SystemC object name for "
63 "PciDeviceQueryInterface and "
64 "BaseAddressRegisterQueryInterface implementor");
65 return;
66 }
67
68 iface::PciDeviceQueryInterface *pci =
69 dynamic_cast<iface::PciDeviceQueryInterface *>(obj);
70 iface::BaseAddressRegisterQueryInterface *bar =
71 dynamic_cast<iface::BaseAddressRegisterQueryInterface *>(obj);
72 this->connect(pci, bar, this->obj());
73 }
74 static conf_class_t *registerGasketClass(const char* class_name) {
75 conf_class_t *cls = SIM_get_class(class_name);
76 if (SIM_clear_exception() == SimExc_No_Exception) {
77 return cls;
78 }
79
80 auto new_cls = make_class<PciGasketClass>(
81 class_name,
82 "model of SystemC compsite pci gasket class",
83 "Class for binding Simics interface to SystemC Pci Gasket.",
84 Sim_Class_Kind_Pseudo);
85
86 new_cls->add(Attribute("device", "s",
87 "Name of the SystemC object that implements the"
88 " PciDeviceQueryInterface and"
89 " BaseAddressRegisterQueryInterface."
90 "All TLM2 sockets returned by the object must"
91 " have BUSWIDTH = 32 and TYPES ="
92 " tlm::tlm_base_protocol_types",
93 ATTR_CLS_VAR(PciGasketClass,
94 sc_pci_device_name_),
95 Sim_Attr_Required));
96 new_cls->add(Attribute("simulation", "o",
97 "Simics object implementing the SystemC"
98 " Simulation interface.",
99 ATTR_CLS_VAR(PciGasketClass, simulation_ref_),
100 Sim_Attr_Required));
101 PciGasketBase::initClassInternal<PciGasketClass>(new_cls.get());
102
103 return *new_cls.get();
104 }
105
106 private:
107 std::string sc_pci_device_name_;
108 ConfObjectRef simulation_ref_;
109};
