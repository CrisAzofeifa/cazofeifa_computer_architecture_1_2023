63 ExtensionSetter<TPAYLOAD,

65 return true;
66 }
67
68 return false;

70 bool addrSpaceLookUp(AttrDictParser *parser, simics::types::addr_space_t *v,
71 std::string key) {
72 uint64_t space = 0;
73 if (!parser->lookUp("space", &space))
74 return false;
75
76 if (space > types::Sim_Addr_Space_Memory) {
77 parser->reportError("%s must be in range [%i - %i]",
78 key.c_str(),
79 types::Sim_Addr_Space_Conf,
80 types::Sim_Addr_Space_Memory);
81 return false;
82 }
83
84 *v = static_cast<types::addr_space_t>(space);
85 return true;
86 }
